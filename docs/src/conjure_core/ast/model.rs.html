19/// Represents a computational model containing variables, constraints, and a shared context.
20///
21/// The `Model` struct holds a set of variables and constraints for manipulating and evaluating symbolic expressions.
22///
23/// # Fields
24/// - `constraints`:
25///   - Type: `Vec<Expression>`
26///   - Represents the logical constraints applied to the model's variables.
27///   - Can be a single constraint or a combination of various expressions, such as logical operations (e.g., `AND`, `OR`),
28///     arithmetic operations (e.g., `SafeDiv`, `UnsafeDiv`), or specialized constraints like `SumEq`.
29///
30/// - `context`:
31///   - Type: `Arc<RwLock<Context<'static>>>`
32///   - A shared object that stores global settings and state for the model.
33///   - Can be safely read or changed by multiple parts of the program at the same time, making it good for multi-threaded use.
34///
35/// # Usage
36/// This struct is typically used to:
37/// - Define a set of variables and constraints for rule-based evaluation.
38/// - Have transformations, optimizations, and simplifications applied to it using a set of rules.
39#[derive(Derivative, Clone, Debug, Serialize, Deserialize)]
40#[derivative(PartialEq, Eq)]
41pub struct Model {
42    /// Top level constraints. This should be a `Expression::Root`.
43    constraints: Box<Expression>,
44
45    symbols: SymbolTable,
46
47    #[serde(skip)]
48    #[derivative(PartialEq = "ignore")]
49    pub context: Arc<RwLock<Context<'static>>>,
50}
51
52impl Model {
53    /// Creates a new model.
54    pub fn new(
55        symbols: SymbolTable,
56        constraints: Vec<Expression>,
57        context: Arc<RwLock<Context<'static>>>,
58    ) -> Model {
59        Model {
60            symbols,
61            constraints: Box::new(Expression::Root(Metadata::new(), constraints)),
62            context,
63        }
64    }
65
66    pub fn new_empty(context: Arc<RwLock<Context<'static>>>) -> Model {
67        Model::new(Default::default(), Vec::new(), context)
68    }
69
70    /// The global symbol table for this model.
71    pub fn symbols(&self) -> &SymbolTable {
72        &self.symbols
73    }
74
75    /// The global symbol table for this model, as a mutable reference.
76    pub fn symbols_mut(&mut self) -> &mut SymbolTable {
77        &mut self.symbols
78    }
79
80    pub fn get_constraints_vec(&self) -> Vec<Expression> {
81        match *self.constraints {
82            Expression::Root(_, ref exprs) => exprs.clone(),
83            ref e => {
84                bug!(
85                    "get_constraints_vec: unexpected top level expression, {} ",
86                    e
87                );
88            }
89        }
90    }
91
92    pub fn set_constraints(&mut self, constraints: Vec<Expression>) {
93        self.constraints = Box::new(Expression::Root(Metadata::new(), constraints));
94    }
95
96    pub fn set_context(&mut self, context: Arc<RwLock<Context<'static>>>) {
97        self.context = context;
98    }
99
100    pub fn add_constraint(&mut self, expression: Expression) {
101        // TODO (gs248): there is no checking whatsoever
102        // We need to properly validate the expression but this is just for testing
103        let mut constraints = self.get_constraints_vec();
104        constraints.push(expression);
105        self.set_constraints(constraints);
106    }
107
108    pub fn add_constraints(&mut self, expressions: Vec<Expression>) {
109        let mut constraints = self.get_constraints_vec();
110        constraints.extend(expressions);
111        self.set_constraints(constraints);
112    }
113}

115// At time of writing (03/02/2025), the Uniplate derive macro doesn't like the lifetimes inside
116// context, and we do not yet have a way of ignoring this field.
117impl Uniplate for Model {
118    fn uniplate(&self) -> (Tree<Self>, Box<dyn Fn(Tree<Self>) -> Self>) {
119        // Model contains no sub-models.
120        let self2 = self.clone();
121        (Tree::Zero, Box::new(move |_| self2.clone()))
122    }
123}
124
125// TODO: replace with derive macro when possible.
126impl Biplate<Expression> for Model {
127    fn biplate(&self) -> (Tree<Expression>, Box<dyn Fn(Tree<Expression>) -> Self>) {
128        let (symtab_tree, symtab_ctx) = self.symbols.biplate();
129        let (constraints_tree, constraints_ctx) = self.constraints.biplate();
130
131        let tree = Tree::Many(VecDeque::from([symtab_tree, constraints_tree]));
132
133        let self2 = self.clone();
134        let ctx = Box::new(move |tree| {
135            let Tree::Many(fields) = tree else {
136                panic!("number of children changed!");
137            };
138
139            let mut self3 = self2.clone();
140            self3.symbols = (symtab_ctx)(fields[0].clone());
141            self3.constraints = Box::new((constraints_ctx)(fields[1].clone()));
142            self3
143        });
144
145        (tree, ctx)
146    }
147}

149impl Display for Model {
150    #[allow(clippy::unwrap_used)] // [rustdocs]: should only fail iff the formatter fails
151    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
152        for (name, _) in self.symbols().iter_value_letting() {
153            writeln!(
154                f,
155                "{}",
156                pretty_value_letting_declaration(self.symbols(), name).unwrap()
157            )?;
158        }
159        for (name, _) in self.symbols().iter_domain_letting() {
160            writeln!(
161                f,
162                "{}",
163                pretty_domain_letting_declaration(self.symbols(), name).unwrap()
164            )?;
165        }
166
167        for (name, _) in self.symbols().iter_var() {
168            writeln!(
169                f,
170                "find {}",
171                pretty_variable_declaration(self.symbols(), name).unwrap()
172            )?;
173        }
174
175        writeln!(f, "\nsuch that\n")?;
176
177        writeln!(
178            f,
179            "{}",
180            pretty_expressions_as_top_level(&self.get_constraints_vec())
181        )?;
182
183        Ok(())
184    }
